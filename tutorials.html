const nameArray = ["Alex", "Keith", "Hannah", "Jack", "Gabriella", "Steven"]; var arrayCount = 0; var longestName = ""; function arrayCycle() { if (arrayCount < nameArray.length) { if (nameArray[arrayCount].length > longestName.length) { longestName.length = nameArray[arrayCount].length; longestName = nameArray[arrayCount]; } arrayCount = arrayCount + 1; arrayCycle(); } } arrayCycle(); console.log(`The longest name was ${longestName} with ${longestName.length} letters!`);
